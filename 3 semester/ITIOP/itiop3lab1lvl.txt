Sub InsertTable()
    table_name = " " + InputBox("Tabel Name:")
    table_rows = InputBox("Table Rows:")
    table_cols = InputBox("Table Columns:")
    
    Selection.InsertCaption Label:="Таблица", TitleAutoText:="table_name", _
         Title:=table_name, Position:=wdCaptionPositionAbove, ExcludeLabel:=0
    
    ActiveDocument.Tables.Add Range:=Selection.Range, NumRows:=table_rows, NumColumns:=table_cols, DefaultTableBehavior:=wdWord9TableBehavior, AutoFitBehavior:=wdAutoFitFixed
    Selection.SelectColumn
    ListGalleries(wdNumberGallery).ListTemplates(1).Name = ""
    Selection.Range.ListFormat.ApplyListTemplateWithLevel ListTemplate:= _
        ListGalleries(wdNumberGallery).ListTemplates(1), ContinuePreviousList:= _
        False, ApplyTo:=wdListApplyToWholeList, DefaultListBehavior:= _
        wdWord10ListBehavior
        
End Sub
